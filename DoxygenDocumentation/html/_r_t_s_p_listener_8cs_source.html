323 
324  switch (currentByte)

327  // the read is blocking, so if we got -1 it is because the client close;
328  currentReadingState = ReadingState.End;
329  needMoreChar = false;

333  buffer.Clear();

378  currentMessage.Data.Length - byteReaden);

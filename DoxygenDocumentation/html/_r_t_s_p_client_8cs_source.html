528  System.Array.Copy(e.Message.Data, rtp_payload_start, rtp_payload, 0, rtp_payload.Length); // copy payload
529 
530  List<byte[]> audio_frames = g711Payload.Process_G711_RTP_Packet(rtp_payload, rtp_marker);
531 
532  if (audio_frames == null) {
533  // some error
534  } else {
535  // Write the audio frames to the file
536  if (Received_G711 != null) {
537  Received_G711(audio_codec, audio_frames);
538  }
539  }
540  }
541  else if (data_received.Channel == audio_data_channel
542  && rtp_payload_type == audio_payload
543  && audio_codec.Equals("AMR")) {
544  // AMR
545  byte[] rtp_payload = new byte[e.Message.Data.Length - rtp_payload_start]; // payload with RTP header removed
546  System.Array.Copy(e.Message.Data, rtp_payload_start, rtp_payload, 0, rtp_payload.Length); // copy payload
547 
548  List<byte[]> audio_frames = amrPayload.Process_AMR_RTP_Packet(rtp_payload, rtp_marker);
549 
550  if (audio_frames == null) {
551  // some error
552  } else {
553  // Write the audio frames to the file
554  if (Received_AMR != null) {
555  Received_AMR(audio_codec, audio_frames);
556  }
557  }
558  }
559  else if (data_received.Channel == audio_data_channel
560  && rtp_payload_type == audio_payload
561  && audio_codec.Equals("MPEG4-GENERIC")
562  && aacPayload != null)
563  {
564  // AAC
565  byte[] rtp_payload = new byte[e.Message.Data.Length - rtp_payload_start]; // payload with RTP header removed
566  System.Array.Copy(e.Message.Data, rtp_payload_start, rtp_payload, 0, rtp_payload.Length); // copy payload
567 
568  List<byte[]> audio_frames = aacPayload.Process_AAC_RTP_Packet(rtp_payload, rtp_marker);
569 
570  if (audio_frames == null) {
571  // some error
572  } else {
573  // Write the audio frames to the file
574  if (Received_AAC != null) {
575  Received_AAC(audio_codec, audio_frames, aacPayload.ObjectType, aacPayload.FrequencyIndex, aacPayload.ChannelConfiguration);
576  }
577  }
578  }
579  else if (data_received.Channel == video_data_channel && rtp_payload_type == 26) {
580  _logger.Warn("No parser has been written for JPEG RTP packets. Please help write one");
581  return; // ignore this data
582  }
583  else {
584  _logger.Warn("No parser for RTP payload " + rtp_payload_type);
585  }
586  }

590  // RTSP Messages are OPTIONS, DESCRIBE, SETUP, PLAY etc
591  private void Rtsp_MessageReceived(object sender, Rtsp.RtspChunkEventArgs e)
592  {
593  Rtsp.Messages.RtspResponse message = e.Message as Rtsp.Messages.RtspResponse;
594 
595  _logger.Debug("Received RTSP Message " + message.OriginalRequest.ToString());
596 
597  // If message has a 401 - Unauthorised Error, then we re-send the message with Authorization
598  // using the most recently received 'realm' and 'nonce'
599  if (message.IsOk == false) {
600  _logger.Debug("Got Error in RTSP Reply " + message.ReturnCode + " " + message.ReturnMessage);
601 
602  if (message.ReturnCode == 401 && (message.OriginalRequest.Headers.ContainsKey(RtspHeaderNames.Authorization)==true)) {
603  // the authorization failed.
604  Stop();
605  return;
606  }
607 
608  // Check if the Reply has an Authenticate header.
609  if (message.ReturnCode == 401 && message.Headers.ContainsKey(RtspHeaderNames.WWWAuthenticate)) {
610 
611  // Process the WWW-Authenticate header
612  // EG: Basic realm="AProxy"
613  // EG: Digest realm="AXIS_WS_ACCC8E3A0A8F", nonce="000057c3Y810622bff50b36005eb5efeae118626a161bf", stale=FALSE
614  // EG: Digest realm="IP Camera(21388)", nonce="534407f373af1bdff561b7b4da295354", stale="FALSE"
615 
616  String www_authenticate = message.Headers[RtspHeaderNames.WWWAuthenticate];
617  String auth_params = "";
618 
619  if (www_authenticate.StartsWith("basic",StringComparison.InvariantCultureIgnoreCase)) {
620  auth_type = "Basic";
621  auth_params = www_authenticate.Substring(5);
622  }
623  if (www_authenticate.StartsWith("digest",StringComparison.InvariantCultureIgnoreCase)) {
624  auth_type = "Digest";
625  auth_params = www_authenticate.Substring(6);
626  }
627 
628  string[] items = auth_params.Split(new char[] { ',' }); // NOTE, does not handle Commas in Quotes
629 
630  foreach (string item in items) {
631  // Split on the = symbol and update the realm and nonce
632  string[] parts = item.Trim().Split(new char[] {'='},2); // max 2 parts in the results array
633  if (parts.Count() >= 2 && parts[0].Trim().Equals("realm")) {
634  realm = parts[1].Trim(new char[] {' ','\"'}); // trim space and quotes
635  }
636  else if (parts.Count() >= 2 && parts[0].Trim().Equals("nonce")) {
637  nonce = parts[1].Trim(new char[] {' ','\"'}); // trim space and quotes
638  }
639  }
640 
641  _logger.Debug("WWW Authorize parsed for " + auth_type + " " + realm + " " + nonce);
642  }
643 
644  RtspMessage resend_message = message.OriginalRequest.Clone() as RtspMessage;
645 
646  if (auth_type != null) {
647  AddAuthorization(resend_message,username,password,auth_type,realm,nonce,url);
648  }
649 
650  rtsp_client.SendMessage(resend_message);
651 
652  return;
653 
654  }
655 
656 
657  // If we get a reply to OPTIONS then start the Keepalive Timer and send DESCRIBE
658  if (message.OriginalRequest != null && message.OriginalRequest is Rtsp.Messages.RtspRequestOptions)
659  {
660 
661  // Check the capabilities returned by OPTIONS
662  // The Public: header contains the list of commands the RTSP server supports
663  // Eg DESCRIBE, SETUP, TEARDOWN, PLAY, PAUSE, OPTIONS, ANNOUNCE, RECORD, GET_PARAMETER]}
664  if (message.Headers.ContainsKey(RtspHeaderNames.Public))
665  {
666  string[] parts = message.Headers[RtspHeaderNames.Public].Split(',');
667  foreach (String part in parts) {
668  if (part.Trim().ToUpper().Equals("GET_PARAMETER")) server_supports_get_parameter = true;
669  if (part.Trim().ToUpper().Equals("SET_PARAMETER")) server_supports_set_parameter = true;
670  }
671  }
672 
673  if (keepalive_timer == null)
674  {
675  // Start a Timer to send an Keepalive RTSP command every 20 seconds
676  keepalive_timer = new System.Timers.Timer();
677  keepalive_timer.Elapsed += Timer_Elapsed;
678  keepalive_timer.Interval = 20 * 1000;
679  keepalive_timer.Enabled = true;
680 
681  // Send DESCRIBE
682  Rtsp.Messages.RtspRequest describe_message = new Rtsp.Messages.RtspRequestDescribe();
683  describe_message.RtspUri = new Uri(url);
684  if (auth_type != null) {
685  AddAuthorization(describe_message,username,password,auth_type,realm,nonce,url);
686  }
687  rtsp_client.SendMessage(describe_message);
688  }
689  else
690  {
691  // If the Keepalive Timer was not null, the OPTIONS reply may have come from a Keepalive
692  // So no need to generate a DESCRIBE message
693  // do nothing
694  }
695  }
696 
697 
698  // If we get a reply to DESCRIBE (which was our second command), then prosess SDP and send the SETUP
699  if (message.OriginalRequest != null && message.OriginalRequest is Rtsp.Messages.RtspRequestDescribe)
700  {
701 
702  // Got a reply for DESCRIBE
703  if (message.IsOk == false) {
704  _logger.Debug("Got Error in DESCRIBE Reply " + message.ReturnCode + " " + message.ReturnMessage);
705  return;
706  }
707 
708  // Examine the SDP
709 
710  _logger.Debug(System.Text.Encoding.UTF8.GetString(message.Data));
711 
712  Rtsp.Sdp.SdpFile sdp_data;
713  using (StreamReader sdp_stream = new StreamReader(new MemoryStream(message.Data)))
714  {
715  sdp_data = Rtsp.Sdp.SdpFile.Read(sdp_stream);
716  }
717 
718  // RTP and RTCP 'channels' are used in TCP Interleaved mode (RTP over RTSP)
719  // These are the channels we request. The camera confirms the channel in the SETUP Reply.
720  // But, a Panasonic decides to use different channels in the reply.
721  int next_free_rtp_channel = 0;
722  int next_free_rtcp_channel = 1;
723 
724  // Process each 'Media' Attribute in the SDP (each sub-stream)
725 
726  for (int x = 0; x < sdp_data.Medias.Count; x++)
727  {
728  bool audio = (sdp_data.Medias[x].MediaType == Rtsp.Sdp.Media.MediaTypes.audio);
729  bool video = (sdp_data.Medias[x].MediaType == Rtsp.Sdp.Media.MediaTypes.video);
730 
731  if (video && video_payload != -1) continue; // have already matched a video payload. don't match another
732  if (audio && audio_payload != -1) continue; // have already matched an audio payload. don't match another
733 
734  if (audio || video)
735  {
736 
737  // search the attributes for control, rtpmap and fmtp
738  // (fmtp only applies to video)
739  String control = ""; // the "track" or "stream id"
740  Rtsp.Sdp.AttributFmtp fmtp = null; // holds SPS and PPS in base64 (h264 video)
741  foreach (Rtsp.Sdp.Attribut attrib in sdp_data.Medias[x].Attributs) {
742  if (attrib.Key.Equals("control")) {
743  String sdp_control = attrib.Value;
744  if (sdp_control.ToLower().StartsWith("rtsp://")) {
745  control = sdp_control; //absolute path
746  } else {
747  control = url + "/" + sdp_control; // relative path
748  }
749  if (video) video_uri = new Uri(control);
750  if (audio) audio_uri = new Uri(control);
751  }
752  if (attrib.Key.Equals("fmtp")) {
753  fmtp = attrib as Rtsp.Sdp.AttributFmtp;
754  }
755  if (attrib.Key.Equals("rtpmap")) {
756  Rtsp.Sdp.AttributRtpMap rtpmap = attrib as Rtsp.Sdp.AttributRtpMap;
757 
758  // Check if the Codec Used (EncodingName) is one we support
759  String[] valid_video_codecs = {"H264","H265"};
760  String[] valid_audio_codecs = {"PCMA", "PCMU", "AMR", "MPEG4-GENERIC" /* for aac */}; // Note some are "mpeg4-generic" lower case
761 
762  if (video && Array.IndexOf(valid_video_codecs,rtpmap.EncodingName.ToUpper()) >= 0) {
763  // found a valid codec
764  video_codec = rtpmap.EncodingName.ToUpper();
765  video_payload = sdp_data.Medias[x].PayloadType;
766  }
767  if (audio && Array.IndexOf(valid_audio_codecs,rtpmap.EncodingName.ToUpper()) >= 0) {
768  audio_codec = rtpmap.EncodingName.ToUpper();
769  audio_payload = sdp_data.Medias[x].PayloadType;
770  }
771  }
772  }
773 
774  // Create H264 RTP Parser
775  if (video && video_codec.Contains("H264"))
776  {
777  h264Payload = new Rtsp.H264Payload();
778  }
779 
780  // If the rtpmap contains H264 then split the fmtp to get the sprop-parameter-sets which hold the SPS and PPS in base64
781  if (video && video_codec.Contains("H264") && fmtp != null) {
782  var param = Rtsp.Sdp.H264Parameters.Parse(fmtp.FormatParameter);
783  var sps_pps = param.SpropParameterSets;
784  if (sps_pps.Count() >= 2) {
785  byte[] sps = sps_pps[0];
786  byte[] pps = sps_pps[1];
787  if (Received_SPS_PPS != null) {
788  Received_SPS_PPS(sps,pps);
789  }
790  h264_sps_pps_fired = true;
791  }
792  }
793 
794  // Create H265 RTP Parser
795  if (video && video_codec.Contains("H265"))
796  {
797  // TODO - check if DONL is being used
798  bool has_donl = false;
799  h265Payload = new Rtsp.H265Payload(has_donl);
800  }
801 
802  // If the rtpmap contains H265 then split the fmtp to get the sprop-vps, sprop-sps and sprop-pps
803  // The RFC makes the VPS, SPS and PPS OPTIONAL so they may not be present. In which we pass back NULL values
804  if (video && video_codec.Contains("H265") && fmtp != null)
805  {
806  var param = Rtsp.Sdp.H265Parameters.Parse(fmtp.FormatParameter);
807  var vps_sps_pps = param.SpropParameterSets;
808  if (vps_sps_pps.Count() >= 3)
809  {
810  byte[] vps = vps_sps_pps[0];
811  byte[] sps = vps_sps_pps[1];
812  byte[] pps = vps_sps_pps[2];
813  if (Received_VPS_SPS_PPS != null)
814  {
815  Received_VPS_SPS_PPS(vps,sps, pps);
816  }
817  h265_vps_sps_pps_fired = true;
818  }
819  }
820 
821  // Create AAC RTP Parser
822  // Example fmtp is "96 profile-level-id=1;mode=AAC-hbr;sizelength=13;indexlength=3;indexdeltalength=3;config=1490"
823  // Example fmtp is ""96 streamtype=5;profile-level-id=1;mode=AAC-hbr;sizelength=13;indexlength=3;indexdeltalength=3;config=1210"
824  if (audio && audio_codec.Contains("MPEG4-GENERIC") && fmtp.GetParameter("mode").ToLower().Equals("aac-hbr"))
825  {
826  // Extract config (eg 0x1490 or 0x1210)
827 
828  aacPayload = new Rtsp.AACPayload(fmtp.GetParameter("config"));
829  }
830 
831 
832  // Send the SETUP RTSP command if we have a matching Payload Decoder
833  if (video && video_payload == -1) continue;
834  if (audio && audio_payload == -1) continue;
835 
836  RtspTransport transport = null;
837 
838  if (rtp_transport == RTP_TRANSPORT.TCP)
839  {
840  // Server interleaves the RTP packets over the RTSP connection
841  // Example for TCP mode (RTP over RTSP) Transport: RTP/AVP/TCP;interleaved=0-1
842  if (video) {
843  video_data_channel = next_free_rtp_channel;
844  video_rtcp_channel = next_free_rtcp_channel;
845  }
846  if (audio) {
847  audio_data_channel = next_free_rtp_channel;
848  audio_rtcp_channel = next_free_rtcp_channel;
849  }
850  transport = new RtspTransport()
851  {
852  LowerTransport = RtspTransport.LowerTransportType.TCP,
853  Interleaved = new PortCouple(next_free_rtp_channel, next_free_rtcp_channel), // Eg Channel 0 for RTP video data. Channel 1 for RTCP status reports
854  };
855 
856  next_free_rtp_channel += 2;
857  next_free_rtcp_channel += 2;
858  }
859  if (rtp_transport == RTP_TRANSPORT.UDP)
860  {
861  int rtp_port = 0;
862  int rtcp_port = 0;
863  // Server sends the RTP packets to a Pair of UDP Ports (one for data, one for rtcp control messages)
864  // Example for UDP mode Transport: RTP/AVP;unicast;client_port=8000-8001
865  if (video) {
866  video_data_channel = video_udp_pair.data_port; // Used in DataReceived event handler
867  video_rtcp_channel = video_udp_pair.control_port; // Used in DataReceived event handler
868  rtp_port = video_udp_pair.data_port;
869  rtcp_port = video_udp_pair.control_port;
870  }
871  if (audio) {
872  audio_data_channel = audio_udp_pair.data_port; // Used in DataReceived event handler
873  audio_rtcp_channel = audio_udp_pair.control_port; // Used in DataReceived event handler
874  rtp_port = audio_udp_pair.data_port;
875  rtcp_port = audio_udp_pair.control_port;
876  }
877  transport = new RtspTransport()
878  {
879  LowerTransport = RtspTransport.LowerTransportType.UDP,
880  IsMulticast = false,
881  ClientPort = new PortCouple(rtp_port, rtcp_port), // a UDP Port for data (video or audio). a UDP Port for RTCP status reports
882  };
883  }
884  if (rtp_transport == RTP_TRANSPORT.MULTICAST)
885  {
886  // Server sends the RTP packets to a Pair of UDP ports (one for data, one for rtcp control messages)
887  // using Multicast Address and Ports that are in the reply to the SETUP message
888  // Example for MULTICAST mode Transport: RTP/AVP;multicast
889  if (video) {
890  video_data_channel = 0; // we get this information in the SETUP message reply
891  video_rtcp_channel = 0; // we get this information in the SETUP message reply
892  }
893  if (audio) {
894  audio_data_channel = 0; // we get this information in the SETUP message reply
895  audio_rtcp_channel = 0; // we get this information in the SETUP message reply
896  }
897  transport = new RtspTransport()
898  {
899  LowerTransport = RtspTransport.LowerTransportType.UDP,
900  IsMulticast = true
901  };
902  }
903 
904  // Generate SETUP messages
905  Rtsp.Messages.RtspRequestSetup setup_message = new Rtsp.Messages.RtspRequestSetup();
906  setup_message.RtspUri = new Uri(control);
907  setup_message.AddTransport(transport);
908  if (auth_type != null) {
909  AddAuthorization(setup_message,username,password,auth_type,realm,nonce,url);
910  }
911 
912  // Add SETUP message to list of mesages to send
913  setup_messages.Add(setup_message);
914 
915  }
916  }
917  // Send the FIRST SETUP message and remove it from the list of Setup Messages
918  rtsp_client.SendMessage(setup_messages[0]);
919  setup_messages.RemoveAt(0);
920  }
921 
922 
923  // If we get a reply to SETUP (which was our third command), then we
924  // (i) check if the Interleaved Channel numbers have been modified by the camera (eg Panasonic cameras)
925  // (ii) check if we have any more SETUP commands to send out (eg if we are doing SETUP for Video and Audio)
926  // (iii) send a PLAY command if all the SETUP command have been sent
927  if (message.OriginalRequest != null && message.OriginalRequest is Rtsp.Messages.RtspRequestSetup)
928  {
929  // Got Reply to SETUP
930  if (message.IsOk == false) {
931  _logger.Debug("Got Error in SETUP Reply " + message.ReturnCode + " " + message.ReturnMessage);
932  return;
933  }
934 
935  _logger.Debug("Got reply from Setup. Session is " + message.Session);
936 
937  session = message.Session; // Session value used with Play, Pause, Teardown and and additional Setups
938  if(message.Timeout > 0 && message.Timeout > keepalive_timer.Interval / 1000)
939  {
940  keepalive_timer.Interval = message.Timeout * 1000 / 2;
941  }
942 
943  // Check the Transport header
944  if (message.Headers.ContainsKey(RtspHeaderNames.Transport))
945  {
946 
947  RtspTransport transport = RtspTransport.Parse(message.Headers[RtspHeaderNames.Transport]);
948 
949  // Check if Transport header includes Multicast
950  if (transport.IsMulticast)
951  {
952  String multicast_address = transport.Destination;
953  video_data_channel = transport.Port.First;
954  video_rtcp_channel = transport.Port.Second;
955 
956  // Create the Pair of UDP Sockets in Multicast mode
957  video_udp_pair = new Rtsp.UDPSocket(multicast_address, video_data_channel, multicast_address, video_rtcp_channel);
958  video_udp_pair.DataReceived += Rtp_DataReceived;
959  video_udp_pair.Start();
960 
961  // TODO - Need to set audio_udp_pair for Multicast
962  }
963 
964  // check if the requested Interleaved channels have been modified by the camera
965  // in the SETUP Reply (Panasonic have a camera that does this)
966  if (transport.LowerTransport == RtspTransport.LowerTransportType.TCP) {
967  if (message.OriginalRequest.RtspUri == video_uri) {
968  video_data_channel = transport.Interleaved.First;
969  video_rtcp_channel = transport.Interleaved.Second;
970  }
971  if (message.OriginalRequest.RtspUri == audio_uri) {
972  audio_data_channel = transport.Interleaved.First;
973  audio_rtcp_channel = transport.Interleaved.Second;
974  }
975 
976  }
977  }
978 
979 
980  // Check if we have another SETUP command to send, then remote it from the list
981  if (setup_messages.Count > 0) {
982  // send the next SETUP message, after adding in the 'session'
983  Rtsp.Messages.RtspRequestSetup next_setup = setup_messages[0];
984  next_setup.Session = session;
985  rtsp_client.SendMessage(next_setup);
986 
987  setup_messages.RemoveAt(0);
988  }
989 
990  else {
991  // Send PLAY
992  Rtsp.Messages.RtspRequest play_message = new Rtsp.Messages.RtspRequestPlay();
993  play_message.RtspUri = new Uri(url);
994  play_message.Session = session;
995  if (auth_type != null) {
996  AddAuthorization(play_message,username,password,auth_type,realm,nonce,url);
997  }
998  rtsp_client.SendMessage(play_message);
999  }
1000  }
1001 
1002  // If we get a reply to PLAY (which was our fourth command), then we should have video being received
1003  if (message.OriginalRequest != null && message.OriginalRequest is Rtsp.Messages.RtspRequestPlay)
1004  {
1005  // Got Reply to PLAY
1006  if (message.IsOk == false) {
1007  _logger.Debug("Got Error in PLAY Reply " + message.ReturnCode + " " + message.ReturnMessage);
1008  return;
1009  }
1010 
1011  _logger.Debug("Got reply from Play " + message.Command);
1012  }

1017  {

1019  // The ONVIF Standard uses SET_PARAMETER as "an optional method to keep an RTSP session alive"
1020  // RFC 2326 (RTSP Standard) says "GET_PARAMETER with no entity body may be used to test client or server liveness("ping")"
1021 
1022  // This code uses GET_PARAMETER (unless OPTIONS report it is not supported, and then it sends OPTIONS as a keepalive)
1023 
1024

1047  // Generate Basic or Digest Authorization
1048  public void AddAuthorization(RtspMessage message, string username, string password,
1049  string auth_type, string realm, string nonce, string url) {
1050 
1051  if (username == null || username.Length == 0) return;
1052  if (password == null || password.Length == 0) return;
1053  if (realm == null || realm.Length == 0) return;
1054  if (auth_type.Equals("Digest") && (nonce == null || nonce.Length == 0)) return;
1055 
1056  if (auth_type.Equals("Basic")) {
1057  byte[] credentials = System.Text.Encoding.UTF8.GetBytes(username+":"+password);
1058  String credentials_base64 = Convert.ToBase64String(credentials);
1059  String basic_authorization = "Basic " + credentials_base64;
1060 
1061  message.Headers.Add(RtspHeaderNames.Authorization, basic_authorization);
1062 
1063  return;
1064  }
1065  else if (auth_type.Equals("Digest")) {
1066 
1067  string method = message.Method; // DESCRIBE, SETUP, PLAY etc
1068 
1069  MD5 md5 = System.Security.Cryptography.MD5.Create();
